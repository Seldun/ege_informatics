```python
from itertools import *



# Пример 1: Перестановки (permutations)
# Генерирует все возможные упорядоченные комбинации без повторений
"""
('A', 'B', 'C')
('A', 'C', 'B')
('B', 'A', 'C')
('B', 'C', 'A')
('C', 'A', 'B')
('C', 'B', 'A')
"""
for i in permutations("ABC",3): # число не может превышать длину строки
    a = ''.join(i)
    print(a)



# Пример 2: Декартово произведение (product)
# Генерирует все возможные комбинации с повторениями
'''
('A', 'A', 'A', 'A')
('A', 'A', 'A', 'S')
('A', 'A', 'A', 'D')
...
('D', 'D', 'D', 'S')
('D', 'D', 'D', 'D')
'''
for i in product("ASD", repeat=4):
    a = ''.join(i)
    print(a)


# Пример 3: Комбинации (combinations)
# Генерирует все возможные неупорядоченные комбинации без повторений
"""
('A', 'S')
('A', 'D')
('S', 'D')
"""
for i in combinations("ASD", 2):# тоже нельзя превышать
    print(i)

```
